Spiciness degree;
public Burrito($piciness degree) {
this.degree = degree;
public void describe() {
System.out.print(”This burrito is ”);
switch(degree) {
case NOT:
System.out.println(
”not spicy at all.”);

case MILD:
case MEDIUM: System.out.println(”a little hot.”);
break;
case HOT:
case FLAMING:
default:
System.out.println(”maybe too hot.”);
}
}
public static void main($tring[] args) {
Burrito
plain = new Burrito($piciness.NOT),
greenChile = new Burrito($piciness.MEDIUM),
jalapeno = new Burrito($piciness.HOT);
plain.describe();
greenChile.describe();
jalapeno.describe();
}
}

This burrito is not spicy at all.
This burrito is a little hot.
